#include <iostream>
using namespace std;

string dia(int n) {
    switch(n) {
        case 1: return "Domingo";
        case 2: return "Segunda";
        case 3: return "Terca";
        case 4: return "Quarta";
        case 5: return "Quinta";
        case 6: return "Sexta";
        case 7: return "Sabado";
        default: return "Invalido";
    }
}

int main() {
    int n;
    cin >> n;

    cout << dia(n);
}
