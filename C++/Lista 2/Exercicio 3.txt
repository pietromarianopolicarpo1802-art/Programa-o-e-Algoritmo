#include <iostream>
using namespace std;

float masculino(float h) { return 72.7 * h - 58; }
float feminino(float h) { return 62.1 * h - 44.7; }

int main() {
    char sexo;
    float h;

    cin >> sexo >> h;

    if(sexo == 'M')
        cout << masculino(h);
    else
        cout << feminino(h);

    return 0;
}
