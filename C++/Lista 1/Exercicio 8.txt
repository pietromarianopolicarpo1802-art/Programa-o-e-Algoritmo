#include <iostream>
using namespace std;

int main() {
    int numero;
    int somaPar = 0;
    int somaImpar = 0;

    cin >> numero;

    for (int i = 0; i <= numero; i++) {

        if (i % 2 == 0)
            somaPar += i;
        else
            somaImpar += i;
    }

    cout << "Soma dos pares: " << somaPar << endl;
    cout << "Soma dos impares: " << somaImpar << endl;

    return 0;
}
