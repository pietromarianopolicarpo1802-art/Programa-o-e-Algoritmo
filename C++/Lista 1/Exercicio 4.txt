#include <iostream>
using namespace std;

int main() {
    int numero;
    int menor, maior;

    for (int i = 0; i < 10; i++) {
        cin >> numero;

        if (i == 0) {
            menor = numero;
            maior = numero;
        }

        if (numero < menor)
            menor = numero;

        if (numero > maior)
            maior = numero;
    }

    cout << "Menor: " << menor << endl;
    cout << "Maior: " << maior << endl;

    return 0;
}
