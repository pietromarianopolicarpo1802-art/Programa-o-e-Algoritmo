#include <iostream>
#include <cstdlib>
#include <ctime>

using namespace std;

int main() {
    srand(time(0));

    int secreto = rand() % 100 + 1;
    int palpite;
    int tentativas = 0;

    do {
        cout << "Digite um palpite: ";
        cin >> palpite;

        tentativas++;

        if (palpite > secreto)
            cout << "Muito alto!" << endl;
        else if (palpite < secreto)
            cout << "Muito baixo!" << endl;
        else
            cout << "Correto!" << endl;

    } while (palpite != secreto);

    cout << "Tentativas: " << tentativas;

    return 0;
}
