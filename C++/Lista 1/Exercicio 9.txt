#include <iostream>
using namespace std;

int main() {
    int limite;

    cin >> limite;

    int a = 0;
    int b = 1;

    cout << a << " " << b << " ";

    while (true) {
        int proximo = a + b;

        if (proximo > limite)
            break;

        cout << proximo << " ";

        a = b;
        b = proximo;
    }

    return 0;
}
