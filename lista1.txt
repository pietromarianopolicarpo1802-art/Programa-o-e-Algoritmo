int main() {

    int numero;
    bool primo = true;

    cout << "Me fale o numero que direi se ele é primo: ";
    cin >> numero;

    if (numero < 2) {
        primo = false;
    } else {

        for (int i = 2; i < numero; i++) {

            if (numero % i == 0) {
                primo = false;
                break;
            }
        } 
    }

    if (primo) {
        cout << "O numero é primo";
    } else {
        cout << "O numero nao é primo";
    }

    return 0;
}#include <iostream>

int main() {

    int limite;
    int a = 0, b = 1, proximo;

    cout << "Digite um numero entre 50 e 100: ";
    cin >> limite;

    if(limite < 50 || limite > 100) {
        cout << "Numero invalido!";
    }
    else {

        cout << "Sequencia de Fibonacci:\n";

        while(a <= limite) {
            cout << a << " ";

            proximo = a + b;
            a = b;
            b = proximo;
        }
    }

    return 0;
}
