#include <iostream>
using namespace std;

int main() {
    int v1[10], v2[10], soma=0;

    for(int i=0;i<10;i++) {
        cin >> v1[i];
        v2[i] = -v1[i];
    }

    for(int i=0;i<10;i++) {
        if(v1[i]>0) soma += v1[i];
        if(v2[i]>0) soma += v2[i];
    }

    cout << soma;
}
