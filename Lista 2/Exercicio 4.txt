#include <iostream>
using namespace std;

int soma(int a, int b) { return a + b; }
int sub(int a, int b) { return a - b; }
int mult(int a, int b) { return a * b; }
float divi(int a, int b) { return (float)a / b; }

int main() {
    int a, b;
    cin >> a >> b;

    cout << soma(a,b) << endl;
    cout << sub(a,b) << endl;
    cout << mult(a,b) << endl;
    cout << divi(a,b);
}
