#include <iostream>
using namespace std;

bool vogal(char c){
    c=tolower(c);
    return c=='a'||c=='e'||c=='i'||c=='o'||c=='u';
}

int main(){
    char v[20];
    int cont[5]={0};

    for(int i=0;i<20;i++){
        cin>>v[i];
        switch(tolower(v[i])){
            case 'a': cont[0]++; break;
            case 'e': cont[1]++; break;
            case 'i': cont[2]++; break;
            case 'o': cont[3]++; break;
            case 'u': cont[4]++; break;
        }
    }

    for(int i=0;i<5;i++) cout<<cont[i]<<" ";
}
